\begin{bmatrix} 
    2 & 0 & 1 & 4 \\

\begin{bmatrix} 
    2 & 0 & 1 & 4 \\
    0 & 2 & -\frac{11}{2} & -8 \\
    0 & 0 & \frac{25}{4} & 8 \\
